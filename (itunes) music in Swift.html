Add music to your app

  1. get an mp3 file, and drag it into your project

  2. import AVFoundation

     import AVFoundation

  3. create an audio player

     var  AudioPlayer = AVAudioPlayer()

  4. insert the following code to make your music run with repeats probably- change music url/filename

     let AssortedMusics = NSURL(fileURLWithPath: 	Bundle.main.path(forResource: "Soft-lounge-background-track", ofType: "mp3")!)

 try? AVAudioSession.sharedInstance().setCategory(AVAudioSession.Category.ambient)

 try? AVAudioSession.sharedInstance().setActive(true)

 AudioPlayer.volume = 3

 AudioPlayer = try! AVAudioPlayer(contentsOf: AssortedMusics as URL)

 AudioPlayer.prepareToPlay()

 AudioPlayer.numberOfLoops = -1

 AudioPlayer.play()
